find_path (json_INCLUDE_DIR json.hpp HINTS "${json_DIR}")

if (NOT json_INCLUDE_DIR)
  message(FATAL_ERROR "JSON package not found.")
endif()

include_directories(
  ${json_INCLUDE_DIR}
)

add_library(gltf
  internal_util.cc
  internal_util.h
  load.cc
  load.h
  memory_stream.cc
  memory_stream.h
  message.cc
  message.h
  messages.inl
  stream.cc
  stream.h
  validate.cc
  validate.h
  cache.cc
  cache.h
  disk_stream.cc
  disk_stream.h
  disk_util.cc
  disk_util.h
  glb_stream.cc
  glb_stream.h
  gltf.cc
  gltf.h
  image_parsing.cc
  image_parsing.h
)

file(GLOB GLTF_HEADERS "*.h")
set_target_properties(gltf PROPERTIES PUBLIC_HEADER "${GLTF_HEADERS}")

install(TARGETS gltf EXPORT ufglib DESTINATION lib/ufg PUBLIC_HEADER DESTINATION include/ufg/gltf)
